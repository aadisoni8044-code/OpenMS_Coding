# Dynamic Game Logic in OpenMS 1.1

function start_game(player_name)
    machine("Welcome, ", player_name)
    health = 100
    score  = 0

    while health > 0
        enemy_damage = random_number(10, 50)
        machine("Enemy attacked! Damage: ", enemy_damage)
        health = health - enemy_damage
        score  = score + 10
        delay(1)

start_game("Hero")
